### 1. 准确性指标 (Accuracy Metrics)

#### 1.1 基础准确率指标
```python
class AccuracyMetrics:
    def calculate_accuracy_metrics(self, predictions, ground_truth):
        """
        计算基础准确率指标
        
        Args:
            predictions: 智能体预测结果
            ground_truth: 人工标注的真实标签
            
        Returns:
            dict: 包含准确率、精确率、召回率、F1分数
        """
        from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score
        
        accuracy = accuracy_score(ground_truth, predictions)
        precision = precision_score(ground_truth, predictions, average='weighted')
        recall = recall_score(ground_truth, predictions, average='weighted')
        f1 = f1_score(ground_truth, predictions, average='weighted')
        
        return {
            'accuracy': accuracy,
            'precision': precision,
            'recall': recall,
            'f1_score': f1
        }
```

#### 1.2 违规类型细分准确率
```python
class ViolationTypeMetrics:
    def evaluate_by_violation_type(self, predictions, ground_truth, violation_types):
        """
        按违规类型评估准确率
        
        违规类型:
        - 虚构原价 (Fictitious Original Price)
        - 虚假折扣 (False Discount) 
        - 要素缺失 (Missing Elements)
        - 价格误导 (Price Misleading)
        """
        violation_metrics = {}
        
        for violation_type in set(violation_types):
            mask = [vt == violation_type for vt in violation_types]
            type_predictions = [p for p, m in zip(predictions, mask) if m]
            type_ground_truth = [gt for gt, m in zip(ground_truth, mask) if m]
            
            if len(type_predictions) > 0:
                accuracy = accuracy_score(type_ground_truth, type_predictions)
                violation_metrics[violation_type] = {
                    'count': len(type_predictions),
                    'accuracy': accuracy
                }
        
        return violation_metrics
```

### 2. 一致性指标 (Consistency Metrics)

#### 2.1 Fleiss' Kappa 一致性系数
```python
class ConsistencyMetrics:
    def calculate_fleiss_kappa(self, agent_results, base_model_results):
        """
        计算Fleiss' Kappa系数，衡量智能体与基座模型的一致性
        
        Kappa解释:
        - < 0.20: 轻微一致
        - 0.21-0.40: 一般一致  
        - 0.41-0.60: 中等一致
        - 0.61-0.80: 高度一致
        - 0.81-1.00: 几乎完全一致
        """
        import numpy as np
        from sklearn.metrics import cohen_kappa_score
        
        # 直接使用Cohen's Kappa计算两模型一致性
        kappa = cohen_kappa_score(agent_results, base_model_results)
        
        # 如果需要计算多评估者一致性，使用Fleiss' Kappa
        def fleiss_kappa(ratings):
            """计算Fleiss' Kappa"""
            n_subjects = len(ratings)
            n_categories = len(set(ratings[0]))
            n_raters = len(ratings[0])
            
            # 构建评级矩阵
            rating_matrix = np.zeros((n_subjects, n_categories))
            for i, ratings_i in enumerate(ratings):
                for rating in ratings_i:
                    rating_matrix[i][rating] += 1
            
            # 计算总体一致性
            p_i = np.sum(rating_matrix * (rating_matrix - 1), axis=1) / (n_raters * (n_raters - 1))
            p_bar = np.mean(p_i)
            
            # 计算期望一致性
            p_j = np.sum(rating_matrix, axis=0) / (n_subjects * n_raters)
            p_bar_e = np.sum(p_j * p_j)
            
            # 计算Kappa
            kappa = (p_bar - p_bar_e) / (1 - p_bar_e)
            return kappa
        
        return {
            'cohen_kappa': kappa,
            'interpretation': self.interpret_kappa(kappa)
        }
    
    def interpret_kappa(self, kappa):
        """解释Kappa系数"""
        if kappa < 0.20:
            return "轻微一致"
        elif kappa < 0.40:
            return "一般一致"
        elif kappa < 0.60:
            return "中等一致"
        elif kappa < 0.80:
            return "高度一致"
        else:
            return "几乎完全一致"
```

class EfficiencyMetrics:
    def measure_inference_efficiency(self, agent_results, base_model_results):
        """
        测量推理效率指标
        """
        import time
        
        # 响应时间分析
        agent_response_times = [result['response_time'] for result in agent_results]
        base_response_times = [result['response_time'] for result in base_model_results]
        
        # 吞吐量分析
        agent_throughput = [1/time for time in agent_response_times]
        base_throughput = [1/time for time in base_response_times]
        
        return {
            'response_time': {
                'agent_mean': np.mean(agent_response_times),
                'agent_std': np.std(agent_response_times),
                'base_mean': np.mean(base_response_times),
                'base_std': np.std(base_response_times)
            },
            'throughput': {
                'agent_mean': np.mean(agent_throughput),
                'base_mean': np.mean(base_throughput)
            },
            'efficiency_improvement': np.mean(agent_throughput) / np.mean(base_throughput)
        }

class InterpretabilityMetrics:
    def evaluate_reasoning_chain(self, agent_results):
        """
        评估推理链的完整性和逻辑性
        """
        def analyze_cot_steps(cot_response):
            """分析CoT推理步骤"""
            steps = cot_response.split('\n')
            reasoning_steps = [step for step in steps if step.strip()]
            
            # 检查是否包含关键要素
            has_law_reference = any('《' in step and '》' in step for step in reasoning_steps)
            has_fact_analysis = any('事实' in step or '数据' in step for step in reasoning_steps)
            has_conclusion = any('结论' in step or '判定' in step for step in reasoning_steps)
            
            return {
                'step_count': len(reasoning_steps),
                'has_law_reference': has_law_reference,
                'has_fact_analysis': has_fact_analysis,
                'has_conclusion': has_conclusion,
                'completeness_score': (has_law_reference + has_fact_analysis + has_conclusion) / 3
            }
        
        reasoning_analysis = []
        for result in agent_results:
            if 'cot_reasoning' in result:
                analysis = analyze_cot_steps(result['cot_reasoning'])
                reasoning_analysis.append(analysis)
        
        avg_completeness = np.mean([analysis['completeness_score'] for analysis in reasoning_analysis])
        avg_step_count = np.mean([analysis['step_count'] for analysis in reasoning_analysis])
        
        return {
            'reasoning_completeness': avg_completeness,
            'average_steps': avg_step_count,
            'law_reference_rate': np.mean([analysis['has_law_reference'] for analysis in reasoning_analysis]),
            'fact_analysis_rate': np.mean([analysis['has_fact_analysis'] for analysis in reasoning_analysis]),
            'conclusion_rate': np.mean([analysis['has_conclusion'] for analysis in reasoning_analysis])
        }
```

class BusinessMetrics:
    def evaluate_business_impact(self, agent_results, base_results, violation_cases):
        """
        评估在真实业务场景中的表现
        """
        # 按严重程度分类
        severe_violations = [vc for vc in violation_cases if vc['severity'] == 'high']
        moderate_violations = [vc for vc in violation_cases if vc['severity'] == 'medium']
        mild_violations = [vc for vc in violation_cases if vc['severity'] == 'low']
        
        results = {}
        
        for severity, cases in [('high', severe_violations), 
                               ('medium', moderate_violations), 
                               ('low', mild_violations)]:
            if cases:
                # 计算该严重程度的检出率
                agent_detections = self.calculate_detection_rate(agent_results, cases)
                base_detections = self.calculate_detection_rate(base_results, cases)
                
                results[severity] = {
                    'case_count': len(cases),
                    'agent_detection_rate': agent_detections,
                    'base_detection_rate': base_detections,
                    'improvement': agent_detections - base_detections
                }
        
        return results
    
    def calculate_detection_rate(self, model_results, violation_cases):
        """计算特定违规类型的检出率"""
        detected = 0
        total = len(violation_cases)
        
        for case in violation_cases:
            # 查找模型结果中是否检测出该违规
            case_detected = any(
                result['case_id'] == case['id'] and 
                result['violation_detected'] == True 
                for result in model_results
            )
            if case_detected:
                detected += 1
        
        return detected / total if total > 0 else 0
```
